class Report extends Model
{
    /**
     * The attributes that are mass assignable.
     *
     * @var array
     */
    protected $fillable = [
        'id', 'work_id', 'file_name' , 'file_date', 'file_description'
    ];

    /**
     * Return work to report.
     *
     * @return Collection
     */
    public function work()
    {
        $this->belongsTo('App\Work');
    }
}
