class Message 
{
    /**
     * Success case.
     *
     * @param  string $operation
     * @param  string $param
     * @param  $genere|null
     * @return string $msg
     */
    public static function success(string $operation, string $param, $genere = null) : string
    {
        if ($genere) {
            $msg = $operation ." de ". $param. " realizada com sucesso!"; 
        } else {
            $msg = $operation ." de ". $param. " realizado com sucesso!";
        }
        return $msg;
    }
}
